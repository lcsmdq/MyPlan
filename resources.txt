def get_users(
    db: Session, 
    skip: int = 0, 
    limit: int = 100,
    is_event_creator: Optional[bool] = None
) -> List[User]:
    """Obtener lista de usuarios con paginaci贸n"""
    query = db.query(User)
    
    if is_event_creator is not None:
        query = query.filter(User.is_event_creator == is_event_creator)
    
    return query.offset(skip).limit(limit).all()

def get_current_user(
     authorization: Optional[str] = Header(None, alias="Authorization"),
    db: Session = Depends(get_db)
) -> User:
    """Dependency para obtener el usuario autenticado desde el token JWT."""
    #  DEBUG - Temporal
    print(f"=== DEBUG AUTH ===")
    print(f"Authorization recibido: {authorization}")
    print(f"Tipo: {type(authorization)}")
    if not authorization:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Token de autenticaci贸n requerido",
            headers={"WWW-Authenticate": "Bearer"},
        )
    
    try:
        scheme, token = authorization.split()
        if scheme.lower() != "bearer":
            raise HTTPException(
                status_code=status.HTTP_401_UNAUTHORIZED,
                detail="Esquema de autenticaci贸n inv谩lido"
            )
    except ValueError:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Header de autenticaci贸n malformado"
        )
    
    user_id = verify_token(token)
    user = crud_users.get_user(db, user_id)
    
    if not user:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Usuario no encontrado"
        )
    
    return user
